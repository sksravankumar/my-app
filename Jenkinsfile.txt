pipeline {
    agent any 
    stages {
        stage('----clean----') { 
            steps {			
	    				sh "git clone https://github.com/sksravankumar/my-app/edit/master/my-app.git"
					sh "mvn clean"
            }
        }
        stage('------Test-----') { 
            steps {
					sh "mvn test"
            }
        }
        stage('-----package----') { 
            steps {
					sh "mvn package"
            }
        }
    }
}
