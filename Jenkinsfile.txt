pipeline {
    agent any 
    stages {
        stage('----clean----') { 
            steps {
	    				sh "rm -rf my-app"
	    				sh "git clone git@github.com:sksravankumar/my-app.git"
					mvn -version
            }
        }
        stage('------Test-----') { 
            steps {
					sh "mvn test"
            }
        }
        stage('-----package----') { 
            steps {
					sh "mvn package"
            }
        }
    }
}
